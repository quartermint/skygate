        "github.com/disintegration/imaging"
        "github.com/kolesa-team/go-webp/encoder"

    // Transcoder handles image transcoding to WebP with resize and timeout.
    type Transcoder struct {
        Quality      int           // WebP quality (0-100), default 30
        MaxWidth     int           // Max width in pixels, default 800
        Timeout      time.Duration // Per-image timeout, default 500ms
        MaxSizeBytes int           // Skip images larger than this
        MinSizeBytes int           // Skip images smaller than this (1024 = 1KB)
        sem          chan struct{} // Concurrency limiter
    }

    // NewTranscoder creates a Transcoder from ImageConfig.
    func NewTranscoder(cfg ImageConfig) *Transcoder {
        return &Transcoder{
            Quality:      cfg.Quality,           // 30 per D-01
            MaxWidth:     cfg.MaxWidth,           // 800 per D-01
            Timeout:      time.Duration(cfg.TimeoutMS) * time.Millisecond, // 500ms per D-03
            MaxSizeBytes: cfg.MaxSizeBytes,       // 10MB
            MinSizeBytes: 1024,                   // 1KB minimum -- smaller images get larger after transcoding (Pitfall 2)
            sem:          make(chan struct{}, cfg.ConcurrentLimit), // 4 concurrent per Pitfall 3
        }
    }

    // TranscodeResult holds the transcoded image data, or nil if skipped/failed.
    type TranscodeResult struct {
        Data        []byte
        ContentType string // always "image/webp" if successful
    }

    // Transcode decodes the image from body, resizes to MaxWidth if wider,
    // encodes to WebP at Quality. Returns nil result (not error) on timeout,
    // skip (too small/too large), or if result is larger than original.
    // Per D-03: 500ms timeout, falls back to original.
    // Per D-04: caller should not pass GIF content types here.
    func (t *Transcoder) Transcode(body []byte, contentType string) *TranscodeResult { ... }
    ```

    The Transcode method must:
    1. Check len(body) < MinSizeBytes -> return nil (Pitfall 2: tiny images get larger)
    2. Check len(body) > MaxSizeBytes -> return nil (Pitfall 3: memory safety)
    3. Acquire semaphore slot (t.sem <- struct{}{}, defer release)
    4. Use context.WithTimeout for D-03's 500ms limit
    5. Decode image using `image.Decode(bytes.NewReader(body))` (handles JPEG and PNG via init imports)
    6. If image width > MaxWidth, resize with `imaging.Resize(img, t.MaxWidth, 0, imaging.Lanczos)` (0 height = preserve aspect ratio)
    7. Encode to WebP: `webp.Encode(&buf, img, options)` where options use `encoder.NewLossyEncoderOptions(encoder.PresetDefault, float32(t.Quality))`
    8. If len(webpBytes) >= len(body), return nil (Pitfall 2: transcoded is not smaller)
    9. Return &TranscodeResult{Data: webpBytes, ContentType: "image/webp"}
    10. On timeout (context.DeadlineExceeded), return nil

    package main

    import (
        "bytes"
        "io"
        "log"
        "mime"
        "net/http"

    )

    // HandlerChain orchestrates Content-Type based response transformation.
    type HandlerChain struct {
        transcoder *Transcoder
        minifier   *Minifier
        db         *DB // nil-safe: if nil, logging is skipped
        verbose    bool
    }

    // NewHandlerChain creates a HandlerChain from the given components.
    func NewHandlerChain(transcoder *Transcoder, minifier *Minifier, db *DB, verbose bool) *HandlerChain {
        return &HandlerChain{
            transcoder: transcoder,
            minifier:   minifier,
            db:         db,
            verbose:    verbose,
        }
    }
    ```

    The main method is `HandleResponse(resp *http.Response) *http.Response` which:

    1. Guard: if resp == nil || resp.Body == nil, return resp unchanged
    2. Extract Content-Type via `resp.Header.Get("Content-Type")`
    3. Parse media type via `mime.ParseMediaType(ct)` to get clean mediaType
    4. Route per D-08:
       - `isImage(mediaType)` returns true for `image/jpeg`, `image/png`, `image/webp` -- but NOT `image/gif` (D-04) and NOT `image/svg+xml` (D-05 routes to minifier)
       - `isMinifiable(mediaType)` returns true for `text/html`, `text/css`, `application/javascript`, `text/javascript`, `application/x-javascript`, `image/svg+xml`, `application/json`
       - Everything else: passthrough
    5. For images:
       - Read full body with `io.ReadAll(io.LimitReader(resp.Body, int64(maxImageSize)))`, close resp.Body
       - Call `transcoder.Transcode(body, mediaType)`
       - If result != nil: replace resp.Body, set Content-Type to "image/webp", set Content-Length, delete Content-Encoding, log compression via db.LogCompression
       - If result == nil (skipped/timeout): restore original body and Content-Length
    6. For minifiable content:
       - Read full body, close resp.Body
       - Check Content-Encoding header; call DecompressIfNeeded(body, encoding) per Pitfall 1
       - Call `minifier.Minify(decompressed, mediaType)`
       - Replace resp.Body, set Content-Length to len(minified), delete Content-Encoding header (per Pitfall 1: after decompressing, don't claim it's still compressed)
       - Log compression via db.LogCompression
    7. Always: after replacing body, set `resp.ContentLength = int64(len(newBody))` and `resp.Header.Set("Content-Length", strconv.Itoa(len(newBody)))` per Pitfall 6

    Helper functions:
    ```go
    func isImage(mediaType string) bool {
        // true for image/jpeg, image/png, image/webp
        // false for image/gif (D-04), image/svg+xml (D-05 -> minifier)
        return strings.HasPrefix(mediaType, "image/") &&
            mediaType != "image/gif" &&
            mediaType != "image/svg+xml"
    }

    func isMinifiable(mediaType string) bool {
        switch mediaType {
        case "text/html", "text/css",
            "application/javascript", "text/javascript", "application/x-javascript",
            "image/svg+xml", "application/json":
            return true
        }
        return false
    }

    func extractDomain(resp *http.Response) string {
        if resp.Request != nil && resp.Request.URL != nil {
            return resp.Request.URL.Hostname()
        }
        return "unknown"
    }
    ```
