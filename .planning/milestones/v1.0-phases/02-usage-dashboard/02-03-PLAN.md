    **Step 2: Create sse.go** -- SSE event streaming handler (per D-03, D-14).
    - `func (s *Server) HandleSSE(w http.ResponseWriter, r *http.Request)` -- following RESEARCH.md Pattern 2.
    - Set headers: `Content-Type: text/event-stream`, `Cache-Control: no-cache`, `Connection: keep-alive`.
    - Assert `http.Flusher` interface.
    - Create ticker at `s.cfg.PollIntervalSec` seconds (default 5s per D-11).
    - On each tick, collect data and send 6 named SSE events:
      1. `event: bandwidth` -- calls ReadPerMACCounters(), computes deltas, sends JSON `{ts, devices: {mac: bps}, total_bps}`.
      2. `event: chart-data` -- builds JSON `{labels: [time strings], values: [mbps values]}` from last 60 data points (5 minutes at 5s intervals).
      3. `event: devices` -- generates HTML fragment for device table tbody (per D-02): each row has device name, usage bytes formatted, top domain, horizontal bar width as percentage of max device.
      4. `event: cap-status` -- generates HTML fragment for plan cap progress bar (per D-19): computes usage percentage against plan cap from settings, assigns color class (green <50%, yellow 50-75%, orange 75-90%, red >90%), includes text "X.X GB / Y GB used".
      5. `event: savings` -- generates HTML fragment: `<p class="savings-amount">$X.XX saved this session</p>` (per D-17).
      6. `event: categories` -- calls Pi-hole top domains, categorizes them, sends JSON `{labels: ["Social Media", ...], values: [count, ...]}` for pie chart.
    - Also send `event: alerts` when usage crosses 50%, 75%, 90% thresholds (per D-20). HTML fragment with appropriate alert-warning or alert-danger class.
    - Write each event as `fmt.Fprintf(w, "event: %s\ndata: %s\n\n", name, data)` then `flusher.Flush()`.
    - On `r.Context().Done()`, return (clean up on client disconnect per RESEARCH.md Pitfall 6).
    - Keep a ring buffer of last 60 bandwidth readings for chart history.

    ```go
    package main

    import (
        "context"
        "flag"
        "log"
        "net/http"
        "os"
        "os/signal"
        "syscall"
        "fmt"
        "time"
    )

    const (
        defaultConfigPath = "/data/skygate/dashboard.yaml"
    )

    func main() {
        configPath := flag.String("config", defaultConfigPath, "path to dashboard YAML config")
        flag.Parse()

        log.SetPrefix("[skygate-dashboard] ")
        log.SetFlags(log.Ldate | log.Ltime | log.Lshortfile)

        // Load config
        cfg, err := LoadConfig(*configPath)
        if err != nil {
            log.Fatalf("Failed to load config: %v", err)
        }
        log.Printf("Config loaded: port=%d, poll=%ds, pihole=%s", cfg.Port, cfg.PollIntervalSec, cfg.PiHoleAddress)

        // Initialize SQLite database
        db, err := NewDB(cfg.DBPath)
        if err != nil {
            log.Fatalf("Failed to open database: %v", err)
        }
        defer db.Close()
        log.Printf("Database initialized at %s", cfg.DBPath)

        // Load domain categories
        cats, err := LoadCategories(cfg.CategoriesFile)
        if err != nil {
            log.Printf("WARN: Failed to load categories: %v (using empty map)", err)
            cats = &CategoryMap{} // empty fallback
        }

        // Initialize Pi-hole client
        pihole := NewPiHoleClient(cfg.PiHoleAddress)
        if cfg.PiHolePassword != "" {
            if err := pihole.Authenticate(cfg.PiHolePassword); err != nil {
                log.Printf("WARN: Pi-hole auth failed: %v (some features may be limited)", err)
            }
        }

        // Create server
        srv := NewServer(cfg, db, cats, pihole)

        // Register HTTP routes
        mux := http.NewServeMux()
        // SSE endpoint
        mux.HandleFunc("/api/events", srv.HandleSSE)
        // REST API
        mux.HandleFunc("/api/stats/devices", srv.HandleGetDevices)
        mux.HandleFunc("/api/stats/domains", srv.HandleGetDomains)
        mux.HandleFunc("/api/stats/savings", srv.HandleGetSavings)
        mux.HandleFunc("/api/settings", func(w http.ResponseWriter, r *http.Request) {
            switch r.Method {
            case http.MethodGet:
                srv.HandleGetSettings(w, r)
            case http.MethodPut:
                srv.HandlePutSettings(w, r)
            default:
                http.Error(w, "Method not allowed", http.StatusMethodNotAllowed)
            }
        })
        // Captive portal
        mux.HandleFunc("/api/captive/accept", srv.HandleCaptiveAccept)

        // HTTP server
        httpSrv := &http.Server{
            Addr:         fmt.Sprintf(":%d", cfg.Port),
            Handler:      mux,
            ReadTimeout:  5 * time.Second,
            WriteTimeout: 0, // SSE requires no write timeout
            IdleTimeout:  120 * time.Second,
        }

        // Graceful shutdown
        ctx, cancel := context.WithCancel(context.Background())
        defer cancel()

        sigCh := make(chan os.Signal, 1)
        signal.Notify(sigCh, syscall.SIGINT, syscall.SIGTERM)
        go func() {
            sig := <-sigCh

            cancel()
    ```
