    ```go
    package main

    import (
        "crypto/tls"
        "log"
        "net/http"
        "strings"
        "sync"

        "github.com/elazarl/goproxy"
    )

    // BypassSet holds domains that should skip MITM interception.
    type BypassSet struct {
        exact    map[string]bool // exact domain matches
        wildcard []string        // wildcard patterns like "*.apple.com"
    }

    // NewBypassSet creates a BypassSet from a list of domain strings.
    // Entries starting with "*." are treated as wildcard patterns.
    // All other entries are exact matches.
    func NewBypassSet(domains []string) *BypassSet {
        bs := &BypassSet{exact: make(map[string]bool)}
        for _, d := range domains {
            d = strings.TrimSpace(d)
            if strings.HasPrefix(d, "*.") {
                bs.wildcard = append(bs.wildcard, strings.TrimPrefix(d, "*"))
            } else {
                bs.exact[d] = true
            }
        }
        return bs
    }

    // Contains returns true if the hostname matches an exact or wildcard entry.
    func (bs *BypassSet) Contains(hostname string) bool {
        if bs.exact[hostname] {
            return true
        }
        for _, suffix := range bs.wildcard {
            if strings.HasSuffix(hostname, suffix) {
                return true
            }
        }
        return false
    }

    // stripPort removes the port from a host:port string.
    func stripPort(host string) string {
        if idx := strings.LastIndex(host, ":"); idx != -1 {
            return host[:idx]
        }
        return host
    }

    // memCertStore implements goproxy.CertStore for in-memory leaf certificate
    // caching per D-13. On-the-fly leaf certificates generated by goproxy's MITM
    // are cached here with LRU eviction to avoid regenerating certs for frequently
    // visited domains.
    type memCertStore struct {
        mu       sync.RWMutex
        cache    map[string]*tls.Certificate
        maxSize  int      // LRU eviction threshold
        order    []string // insertion order for LRU eviction
    }

    // newMemCertStore creates a memCertStore with the given max cache size.
    func newMemCertStore(maxSize int) *memCertStore {
        return &memCertStore{
            cache:   make(map[string]*tls.Certificate),
            maxSize: maxSize,
        }
    }

    // Fetch retrieves a cached leaf certificate for the given hostname.
    // Returns nil, nil if not found (goproxy will generate a new one).
    func (s *memCertStore) Fetch(host string, ctx *goproxy.ProxyCtx) (*tls.Certificate, error) {
        s.mu.RLock()
        defer s.mu.RUnlock()
        cert, ok := s.cache[host]
        if !ok {
            return nil, nil
        }
        return cert, nil
    }

    // Store caches a leaf certificate for the given hostname.
    // Evicts the oldest entry when cache exceeds maxSize (LRU).
    func (s *memCertStore) Store(host string, cert *tls.Certificate, ctx *goproxy.ProxyCtx) error {
        s.mu.Lock()
        defer s.mu.Unlock()
        // LRU eviction: if at capacity, remove oldest entry
        if len(s.cache) >= s.maxSize {
            if len(s.order) > 0 {
                oldest := s.order[0]
                s.order = s.order[1:]
                delete(s.cache, oldest)
            }
        }
        s.cache[host] = cert
        s.order = append(s.order, host)
        return nil
    }

    // SetupProxy configures the goproxy proxy server with:
    // - In-memory CertStore for on-the-fly leaf cert caching (D-13)
    // - Conditional MITM: bypass domains get ConnectAccept, others get ConnectMitm (D-06, D-09)
    // - Response handler chain: Content-Type routing to transcoder/minifier (D-07, D-08)
    // Returns the configured *goproxy.ProxyHttpServer.
    func SetupProxy(caCert *tls.Certificate, bypassSet *BypassSet, chain *HandlerChain, verbose bool) *goproxy.ProxyHttpServer {
        proxy := goproxy.NewProxyHttpServer()
        proxy.Verbose = verbose

        // Per D-13: on-the-fly leaf certificate generation with LRU-cached CertStore.
        // goproxy uses this to cache generated leaf certs per domain, avoiding
        // repeated certificate generation for frequently visited sites.
        // 1024 entries covers typical browsing session with ample headroom.
        proxy.CertStore = newMemCertStore(1024)

        // Conditional MITM per D-09
        proxy.OnRequest().HandleConnectFunc(
            func(host string, ctx *goproxy.ProxyCtx) (*goproxy.ConnectAction, string) {
                hostname := stripPort(host)
                if bypassSet.Contains(hostname) {
                    if verbose {
                        log.Printf("BYPASS: %s (cert-pinned/bypass list)", hostname)
                    }
                    return &goproxy.ConnectAction{Action: goproxy.ConnectAccept}, host
                }
                return &goproxy.ConnectAction{
                    Action:    goproxy.ConnectMitm,
                    TLSConfig: goproxy.TLSConfigFromCA(caCert),
                }, host
            })

        // Response transformation pipeline per D-07
        proxy.OnResponse().DoFunc(
            func(resp *http.Response, ctx *goproxy.ProxyCtx) *http.Response {
                return chain.HandleResponse(resp)
            })

        return proxy
    }
    ```

    ```go
    package main

    import (
        "context"
        "flag"
        "fmt"
        "log"
        "net/http"
        "os"
        "os/signal"
        "syscall"
        "time"
    )

    const defaultConfigPath = "/etc/skygate/proxy.yaml"

    func main() {
        configPath := flag.String("config", defaultConfigPath, "path to proxy YAML config")
        flag.Parse()

        log.SetPrefix("[skygate-proxy] ")
        log.SetFlags(log.Ldate | log.Ltime | log.Lshortfile)

        // 1. Load config
        cfg, err := LoadConfig(*configPath)
        if err != nil {
            log.Fatalf("Failed to load config: %v", err)
        }
        log.Printf("Config loaded: listen=%s, image_quality=%d, max_width=%d",
            cfg.ListenAddr, cfg.Image.Quality, cfg.Image.MaxWidth)

        // 2. Load or generate CA certificate (D-11)
        caCert, err := LoadOrGenerateCA(cfg.CACertPath, cfg.CAKeyPath)
        if err != nil {
            log.Fatalf("Failed to load/generate CA: %v", err)
        }
        log.Printf("CA certificate ready at %s", cfg.CACertPath)

        // 3. Load bypass domains (D-09)
        bypassDomains, err := LoadBypassDomains(cfg.BypassDomainsFile)
        if err != nil {
            log.Printf("WARN: Failed to load bypass domains: %v (using empty list)", err)
            bypassDomains = []string{}
        }
        bypassSet := NewBypassSet(bypassDomains)
        log.Printf("Loaded %d bypass domains", len(bypassDomains))

        // 4. Initialize SQLite database (D-10)
        db, err := NewDB(cfg.DBPath)
        if err != nil {
            log.Fatalf("Failed to open database: %v", err)
        }
        defer db.Close()
        log.Printf("Database initialized at %s", cfg.DBPath)

        // 5. Create processing components
        transcoder := NewTranscoder(cfg.Image)
        minifier := NewMinifier(cfg.Minify)
        chain := NewHandlerChain(transcoder, minifier, db, cfg.Verbose)

        // 6. Setup goproxy
        proxy := SetupProxy(caCert, bypassSet, chain, cfg.Verbose)

        // 7. CA cert download endpoint (D-12)
        // Serve CA cert on CADownloadAddr for Phase 5 captive portal to fetch
        go func() {
            mux := http.NewServeMux()
            mux.HandleFunc("/ca.crt", func(w http.ResponseWriter, r *http.Request) {
                w.Header().Set("Content-Type", "application/x-x509-ca-cert")
                w.Header().Set("Content-Disposition", "attachment; filename=\"skygate-ca.crt\"")
                http.ServeFile(w, r, cfg.CACertPath)
            })
            mux.HandleFunc("/health", func(w http.ResponseWriter, r *http.Request) {
                w.Header().Set("Content-Type", "application/json")
                fmt.Fprintf(w, `{"status":"ok","service":"skygate-proxy"}`)
            })
            caServer := &http.Server{
                Addr:         cfg.CADownloadAddr,
                Handler:      mux,
                ReadTimeout:  5 * time.Second,
                WriteTimeout: 10 * time.Second,
            }
            log.Printf("CA download endpoint listening on %s", cfg.CADownloadAddr)
            if err := caServer.ListenAndServe(); err != http.ErrServerClosed {
                log.Printf("CA download server error: %v", err)
            }
        }()

        // 8. Graceful shutdown
        ctx, cancel := context.WithCancel(context.Background())
        defer cancel()

        sigCh := make(chan os.Signal, 1)
        signal.Notify(sigCh, syscall.SIGINT, syscall.SIGTERM)
        go func() {
            sig := <-sigCh
            log.Printf("Received signal %v, shutting down...", sig)
            cancel()
        }()

        // 9. Start proxy server
        proxyServer := &http.Server{
            Addr:         cfg.ListenAddr,
            Handler:      proxy,
            ReadTimeout:  30 * time.Second,
            WriteTimeout: 0, // Proxy needs unlimited write time for large responses
            IdleTimeout:  120 * time.Second,
        }
        log.Printf("Proxy server listening on %s", cfg.ListenAddr)
        go func() {
            <-ctx.Done()
            shutdownCtx, shutdownCancel := context.WithTimeout(context.Background(), 10*time.Second)
            defer shutdownCancel()
            proxyServer.Shutdown(shutdownCtx)
        }()

        if err := proxyServer.ListenAndServe(); err != http.ErrServerClosed {
            log.Fatalf("Proxy server error: %v", err)
        }
        log.Println("Proxy server stopped")
    }
    ```
